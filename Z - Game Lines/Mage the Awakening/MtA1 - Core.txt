/*
================================================================================
== CHECK ARCANUM VS. GNOSIS ====================================================

There's no easy way around this: We need to know the max Arcanum level and 
other Arcanum levels.

0: sheet dbref
1: arcanum being raised
2: amt being raised by

m: max sphere values by gnosis: <max sphere> <other spheres>
a: arcana values, sorted high->low

returns: 1 if the sphere is allowed by Gnosis, 0 otherwise

*/

&.arcanum_check [v( d.dd )]=
	localize( strcat( 
		setq( m, u( .arcanum_check.max, %0 )), 

		setq( a, 
			revwords( sort( 
				iter( lattr( %!/arcanum.* ), 
					add( 
						u( .value, %0, %i0 ), 
						if( strmatch( %1, %i0 ), 
							%2, 
							0 
						)
					)
				)
			))
		),

// check highest to first number in %qm, others to the rest
		cand( 
			lte( first( %qa ), first( %qm )), 
			land( iter( rest( %qa ), lte( %i0, rest( %qm ))))
		)
	))


&.arcanum_check.max [v( d.dd )]=
	case( u( .value, %0, advantage.gnosis ), 
		1,  3 2, 
		2,  3 3, 
		3,  4 3, 
		4,  4 4, 
		5,  5 4, 
		6,  5 5, 
		7,  5 5, 
		8,  5 5, 
		9,  5 5, 
		10, 5 5, 
		#-1
	)

		if( lt( %q9, 5 ), + ), 
		%, or Proximus taking a Blessing 
	)


/*
================================================================================
== CHECK ROTE FOR MAGE VS. PROXIMUS ============================================

	u( .rote.prereq-check, %0, %1, Ectoplasmic Shaping, Death 1 )

The fun of Proximi taking spells as 'Blessings'.

0: sheet dbref
1: rote's instance/type
2: name of this rote (for the Proximus' Dynasty check)
3: <sphere> <min level> (for Mage the check)

*/

// &.rote_check [v( d.dd )]=
&.rote.prereq-check [v( d.dd )]=
	case( u( .value_full, %0, bio.template ), 

// Mage: has proper arcana; can't take as blessing, silly
		Mage, 
		cand( 
			u( .at_least, %0, arcanum.[first( %3 )], rest( %3 )), 
			not( strmatch( %1, blessing ))
		), 

// Proximus: belongs to a dynasty that can take this; taking as blessing
		Proximus, 
		cand( 
			u( v( d.sfp )/f.hastag?.workhorse, 
				rote.[edit( %3, %b, _ )]_(), 
				u( .value_full, %0, bio.dynasty )
			), 
			strmatch( %1, blessing )
		)
	)
